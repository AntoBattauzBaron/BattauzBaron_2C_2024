154static void medirDistanciaTask(void *pvParameter){
155
156 uint16_t distancia = 0;
157
158 while(1){
159
160 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
161
162 distancia = HcSr04ReadDistanceInCentimeters();
163
164 if(distancia>500)
165 {
166 LedOn(LED_1);
167 LedOff(LED_2);
168 LedOff(LED_3);
169 precaución = false;
170 peligro = false;
171 }
172
173 else if(distancia>300 && distancia<500)
174 {
175 LedOn(LED_1);
176 LedOn(LED_2);
177 LedOff(LED_3);
178 precaución = true;
179 peligro = false;
180 UartSendString(UART_CONNECTOR, "Precaución, vehículo cerca \r\n");
181 }
182
183 else if(distancia<300)
184 {
185 LedOn(LED_1);
186 LedOn(LED_2);
187 LedOn(LED_3);
188 peligro = true;
189 precaución = false;
190 UartSendString(UART_CONNECTOR, "Peligro, vehículo cerca \r\n");
191 }
192 control_buzzer();
193 }

200static void medirAceleracionTask(void *pvParameter){
201
202 uint16_t aceleracion_en_x = 0;
203 uint16_t aceleracion_en_y = 0;
204 uint16_t aceleracion_en_z = 0;
205

207 uint16_t voltaje_en_y = 0;
208 uint16_t voltaje_en_z = 0;
209
210 analog_input_config_t aceleracion_x = {
211 .input = CH1,
212 .mode = ADC_SINGLE,
213 .func_p = NULL,
214 .param_p = NULL,
215 .sample_frec =NULL
216 };
217 AnalogInputInit(&aceleracion_x);
218
219 analog_input_config_t aceleracion_y = {
220 .input = CH2,
221 .mode = ADC_SINGLE,
222 .func_p = NULL,
223 .param_p = NULL,
224 .sample_frec =NULL
225 };
226 AnalogInputInit(&aceleracion_y);
227
228 analog_input_config_t aceleracion_z = {
229 .input = CH3,
230 .mode = ADC_SINGLE,
231 .func_p = NULL,
232 .param_p = NULL,
233 .sample_frec =NULL
234 };
235 AnalogInputInit(&aceleracion_z);
236
237 while(1){
238
239 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
240
241 AnalogInputReadSingle(CH1, &voltaje_en_x);
242 AnalogInputReadSingle(CH2, &voltaje_en_y);
243 AnalogInputReadSingle(CH3, &voltaje_en_z);
244
245 aceleracion_en_x = (voltaje_en_x - ORDENADA_AL_ORIGEN_ACELERACION) / PENDIENTE_ACELERACION;
246 aceleracion_en_y = (voltaje_en_y - ORDENADA_AL_ORIGEN_ACELERACION) / PENDIENTE_ACELERACION;
247 aceleracion_en_z = (voltaje_en_z - ORDENADA_AL_ORIGEN_ACELERACION) / PENDIENTE_ACELERACION;
248
249 if((aceleracion_en_x + aceleracion_en_y + aceleracion_en_z) >= UMBRAL_ACCIDENTE)
250 {
251 UartSendString(UART_CONNECTOR, "Caída detectada \r\n");
252 }
253 }
254}
